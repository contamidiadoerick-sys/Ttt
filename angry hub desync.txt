-- ==================== MULTIPLATFORM DRAGGING LOGIC ====================
local dragging = false
local dragInput, dragStart, startPos

local function updateDrag(input)
	local delta = input.Position - dragStart
	local targetPosition = UDim2.new(
		startPos.X.Scale, 
		startPos.X.Offset + delta.X, 
		startPos.Y.Scale, 
		startPos.Y.Offset + delta.Y
	)
	
	-- Keep main frame and its bottom web link bound together seamlessly
	mainFrame.Position = targetPosition
	linkFrame.Position = UDim2.new(targetPosition.X.Scale, targetPosition.X.Offset, targetPosition.Y.Scale, targetPosition.Y.Offset + 185)
end

mainFrame.InputBegan:Connect(function(input)
	if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
		dragging = true
		dragStart = input.Position
		startPos = mainFrame.Position
		
		input.Changed:Connect(function()
			if input.UserInputState == Enum.UserInputState.End then
				dragging = false
			end
		end)
	end
end)

mainFrame.InputChanged:Connect(function(input)
	if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
		dragInput = input
	end
end)

UserInputService.InputChanged:Connect(function(input)
	if input == dragInput and dragging then
		updateDrag(input)
	end
end)

-- Active Position Monitor Loop to catch Anti-cheat updates/Teleports
local function startTrackingPosition(hrp)
    if positionTrackerConnection then positionTrackerConnection:Disconnect() end
    
    -- Monitors the HumanoidRootPart's position dynamically
    positionTrackerConnection = hrp:GetPropertyChangedSignal("Position"):Connect(function()
        if desyncActive and hrp then
            updateServerPositionMarker(hrp.Position)
        end
    end)
end

-- Function to completely hide and reset character immediately
local function respawnCharacterInvisible()
    local char = player.Character
    if char then
        local hrp = char:FindFirstChild("HumanoidRootPart")
        if hrp then
            updateServerPositionMarker(hrp.Position)
            startTrackingPosition(hrp)
        end

        -- Hides every visual part instantly so nothing drops or fractures visibly
        for _, obj in pairs(char:GetDescendants()) do
            if obj:IsA("BasePart") then
                obj.Transparency = 1
                obj.CanCollide = false
            elseif obj:IsA("Decal") or obj:IsA("Texture") then
                obj.Transparency = 1
            elseif obj:IsA("BackgroundBlur") or obj:IsA("GuiObject") or obj:IsA("LayerCollector") then
                -- Leave interface instances unaffected
            elseif obj:IsA("ParticleEmitter") or obj:IsA("Trail") or obj:IsA("Beam") then
                obj.Enabled = false
            elseif obj:IsA("Accoutrement") or obj:IsA("Accessory") then
                obj:Destroy() -- Removes hats/hair instantly
            end
        end
        
        -- Kills the humanoid to trigger the native respawn after hiding parts
        if char:FindFirstChild("Humanoid") then
            char.Humanoid.Health = 0
        end
        task.wait(0.5)
    end
end

local function toggleNoAnim(state)
    local char = player.Character
    if not char then return end
    local hum = char:FindFirstChild("Humanoid")
    local animate = char:FindFirstChild("Animate")
    if not hum or not animate then return end

    if state then
        if noAnimConnection then return end
        
        local idleAnimationObject = animate:FindFirstChild("idle") and animate.idle:FindFirstChildOfClass("Animation")
        
        noAnimConnection = RunService.RenderStepped:Connect(function()
            local playingTracks = hum:GetPlayingAnimationTracks()
            
            for _, track in pairs(playingTracks) do
                if not idleAnimationObject or track.Animation.AnimationId ~= idleAnimationObject.AnimationId then
                    track:Stop()
                end
            end
            
            if idleAnimationObject then
                local idleIsPlaying = false
                for _, track in pairs(playingTracks) do
                    if track.Animation.AnimationId == idleAnimationObject.AnimationId then
                        idleIsPlaying = true
                        track:AdjustSpeed(1)
                        break
                    end
                end
                
                if not idleIsPlaying then
                    local idleTrack = hum:LoadAnimation(idleAnimationObject)
                    idleTrack.Looped = true
                    idleTrack:Play()
                end
            end
        end)
    else
        if noAnimConnection then
            noAnimConnection:Disconnect()
            noAnimConnection = nil
        end
    end
end

player.CharacterAdded:Connect(function(char)

    if hrp and desyncActive then
        startTrackingPosition(hrp)
    end
    
    task.wait(1)
    if noAnimActive then
        toggleNoAnim(true)
    end
end)

local fpsCount = 0
local nextTick = os.clock()

local function getPing()
	local ok, val = pcall(function()
		return math.round(player:GetNetworkPing() * 1000)
	end)
	if ok and type(val) == "number" then return val end
	return 0
end

RunService.RenderStepped:Connect(function()
	fpsCount = fpsCount + 1

	local now = os.clock()
	if now >= nextTick then
		fpsLabel.Text  = "FPS: " .. tostring(fpsCount)
		fpsCount       = 0

		local ping = getPing()
		pingLabel.Text = "Ping: " .. tostring(ping) .. "ms"

		if ping < 80 then
			pingLabel.TextColor3 = Color3.fromRGB(80, 255, 120)
		elseif ping < 150 then
			pingLabel.TextColor3 = Color3.fromRGB(255, 220, 60)
		else
			pingLabel.TextColor3 = Color3.fromRGB(255, 80, 80)
		end

		nextTick = now + 1
	end
end)
